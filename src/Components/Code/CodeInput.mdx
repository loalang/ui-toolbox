# CodeInput

The input version of the `Code` component, basically.

import { Playground } from "../../Playground.tsx";
import { CodeInput } from "./CodeInput.tsx";
import { useState } from "react";
import { Button } from "../Forms/Button.tsx";
import { Label } from "../Typography/TextStyle/Label.tsx";

<Playground>
  {() => {
    const [code, setCode] = useState('env print: "Edit Me!".');
    return <CodeInput language="loa" value={code} onChange={setCode} />;
  }}
</Playground>

The component is reactive, and can have its value set programmatically.

<Playground>
  {() => {
    const [code, setCode] = useState("");
    function onClick() {
      setCode("I just set this!");
    }
    return (
      <div>
        <Button onClick={onClick}>
          <Label>Update Programmatically</Label>
        </Button>
        <CodeInput value={code} onChange={setCode} />
      </div>
    );
  }}
</Playground>

I can be disabled, just like other inputs.

import { BooleanInput } from "../Forms/BooleanInput.tsx";

<Playground>
  {() => {
    const [code, setCode] = useState("Change me?");
    const [isDisabled, setIsDisabled] = useState(true);
    return (
      <div>
        <BooleanInput value={isDisabled} onChange={setIsDisabled}>
          <Label>Disable</Label>
        </BooleanInput>
        <CodeInput isDisabled={isDisabled} value={code} onChange={setCode} />
      </div>
    );
  }}
</Playground>
