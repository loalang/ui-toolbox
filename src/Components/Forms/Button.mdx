import { StringInput } from "./StringInput.tsx";
import { useState } from "react";

I am built to have the same height as input fields, for a harmonious combination.

<Playground>
  {() => {
    const [value, setValue] = useState("");
    function onClick() {
      alert(`Value: ${value}`);
    }
    const style = {
      display: "grid",
      gridGap: "10px",
      gridTemplateColumns: "1fr auto"
    };
    return (
      <div style={style}>
        <StringInput value={value} onChange={setValue} />
        <Button onClick={onClick}>Submit</Button>
      </div>
    );
  }}
</Playground>

Since I don't necessarily care what text is inside of me, I can easily be combined with
another text style.
